#include<cstdlib>
#include<iostream>
#include<vector>
using namespace std;
int main()
{
    int p,n;
    long long max=0;
    long long a=0,b=0;
    cin>>n;
    vector<long long> numbers(n);
    for(int i=0;i<n;i++)
        cin>>numbers[i];
    for(int i=0;i<n;i++)
    {
        if(numbers[i]>a)
            a = numbers[i];
    }
    for(int j=0;j<n;j++)
    {
        if(numbers[j]>b&&numbers[j]<a)
            b = numbers[j];
    }
    max = a*b;
    cout<<max<<"\n";
}
